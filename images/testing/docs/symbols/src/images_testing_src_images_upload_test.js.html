 13 /**
 14  * Handles user uploads of files (typically images). 
 15  * Users can upload from their hard drive or provide a URL.
 16  * 
 17  * @constructor
 18  * 
 19  * @property {Array<PresignedURLPacket>} presignedURLPackets
 20  * Objects returned from the server's S3 request, to be used for uploading file.
 21  * 
 22  * @property {Array<String>} processedImageURLs 
 23  * The URLs to images that have been fully processed and uploaded.
 24  * 
 25  * @property {Array<File>} sourceFileArray 
 26  * Files to be uploaded
 27  * 
 28  * @property {String} sourceURL
 29  * A user-provided URL that points to an image to be uploaded
 30  */
 31 function FileSource() {
 32     this.presignedURLPackets = [];
 33     this.processedImageURLs = [];
 34     this.sourceFileArray = [];
 35     this.sourceURL;
 36 }

 39 /**
 40  * Looks for files uploaded to a file input first, defaults to a URL if none.
 41  * The URL itself defaults to a random image API, so sourceFileArray will always
 42  * contain at least one image file. 
 43  * 
 44  * @returns {Array<PresignedURLPacket>}
 45  */
 46  FileSource.prototype.collectFiles = async function() {
 47     // get image URL if provided
 48     if ($('id_image_url')) {
 49         this.sourceURL = $('id_image_url').value;
 50     } else {
 51         this.sourceURL = "https://picsum.photos/300";
 52     }
 53 
 54     // get source files to be uploaded, if any.
 55     if ($('id_image_upload')) {
 56         this.sourceFileArray = Array.from($('id_image_upload')).files;
 57         return this.sourceFileArray;
 58     } else {
 59         let image = await this.getImageFromURL(this.sourceURL);
 60         this.sourceFileArray = [image];
 61         return this.sourceFileArray;
 62     }
 63 }

 66 /**
 67  * Handles the complete upload cycle for a single file. Retrieves a
 68  * presigned URL, uploads file, and returns the URL of the image.
 69  * @async
 70  * 
 71  * @param {File} file 
 72  * 
 73  * @returns {String} URL of the file where it exists in the S3 bucket.
 74  * 
 75  */
 76 FileSource.prototype.processFile = async function(file) {
 77     const packet = await this.getPresignedURLPacket(file);
 78     const url = await this.uploadFileToS3(packet.file, packet.data, packet.url);
 79     return url;
 80 }

 83 /**
 84  * Takes all the files contained in sourceFileArray and uploads them to S3.
 85  * 
 86  * @async
 87  * 
 88  * @returns {void}
 89  */
 90 FileSource.prototype.processAllFiles = async function() {
 91     let promises = [];
 92     for (file of this.sourceFileArray) {
 93         promises.push(this.processFile(file));
 94     }
 95 
 96     this.processedImageURLs = await Promise.all(promises);
 97 }

121 /**
122  * AJAX request for a presigned URL to be used for uploading a file to S3 bucket.
123  * 
124  * @async
125  * 
126  * @param {File} file 
127  * The file that we are retrieving the presigned URL for in order to upload.
128  * 
129  * @returns {Promise<PresignedURLPacket>} 
130  * A {@link PresignedURLPacket} sent from AWS required for uploading file.
131  */
132 FileSource.prototype.getPresignedURLPacket = function(file) {
133     return new Promise((res, rej) => {
134         const url = document.querySelector('[name=sign_s3_url]').value
135             + "?name=" + file.name + "type=" + file.type;
136 
137         makeFetch(url)
138         .then((response) => {
139             return response.json();
140         })
141         .then((json) => {
142             // return the presigned url packet for uploading
143             res(
144                 {
145                     file: file,       // File object
146                     data: json.data,        // {} containing additional information 
147                     url: json.image_url     // the URL of the file once it's uploaded
148                 }
149             );
150         })
151         .catch((error) => {
152             rej(handleFetchError(error));
153         });
154     });
155 }

158 /**
159  * Simple image retriever. Technically this would work for any file type but it's
160  * intended for images.
161  * 
162  * @source https://newbedev.com/how-to-convert-dataurl-to-file-object-in-javascript
163  * 
164  * @async
165  * 
166  * @param {String} url 
167  * Optional. Points to the image to be retrieved. Defaults to a random image API.
168  * 
169  * @param {String} fileName 
170  * Optional. When uploaded to S3, filename gets randomized to a UUID hex.
171  * 
172  * @param {String} fileType
173  * Optional. MIME-Type.
174  * 
175  * @returns {Promise<File>} A Javascript File object containing the new image.
176  */
177 FileSource.prototype.getImageFromURL = function(
178     url="https://picsum.photos/300",
179     fileName="image.jpg",
180     fileType="image/jpg") {
181     return new Promise((res, rej) => {
182         makeFetch(url)
183         .then((response) => {
184             let buffer = response.arrayBuffer();
185             let file = new File([buffer], fileName, {type: fileType});
186             res(file);
187         })
188         .catch((error) => {
189             rej(error);
190         });
191     })
192 }

195 /**
196  * Takes the information from the presigned URL as well as the File and sends
197  * them to AWS. If the request goes through, this method returns the URL of the
198  * file where it exists in the S3 bucket.
199  * 
200  * @async
201  * 
202  * @param {File} file 
203  * The original source file to be uploaded.
204  * 
205  * @param {S3Data} data 
206  * An object returned by AWS, retrieved when the pre-signed URL was generated.
207  * 
208  * @param {String} url 
209  * The location of the file on the S3 bucket, already determined when the 
210  * presigned URL was generated.
211  * 
212  * @returns {Promise<String>} The same url as in the parameters.
213  */
214 FileSource.prototype.uploadFileToS3 = function(file, data, url) {
215     return new Promise((res, rej) => {
216         const postData = new FormData();
217         // all fields need to be copied into the post body
218         for (key in data.fields) {
219             postData.append(key, data.fields[key]);
220         }
221         postData.append('file', file);
222         
223         makeFetch(data.url, {
224             method: "POST",
225             body: postData
226         })
227         .then(() => {
228             res(url);
229         })
230         .catch((error) => {
231             rej(handleFetchError(error));
232         });
233     });
234 }

252 /**
253  * Using this instead of the built-in fetch() API in order to provide better
254  * error-handling.
255  * 
256  * @param {String} url Points to the server path to use for a fetch() request
257  * @param {FetchOptions} options 
258  * @returns {Response}
259  */
260 function makeFetch(url, options={}) {
261     /*
262     options = { method:, body:, headers:, etc.} (from vanilla fetch() api)
263     */
264     return new Promise((res, rej) => {
265         fetch(url, options)
266         .then((response) => {
267             if (response.status >= 200 && response.status <= 299) {
268                 // request is ok, resolve response
269                 res(response);
270             }
271             else {
272                 throw new Error(response.statusText);
273             }
274         })
275         .catch((error) => {
276             rej(error);
277         })
278     });
279 }

282 /**
283  * Handles server error by either updating the UI or logging.
284  * @param {Error} error 
285  * @returns {Error}
286  */
287 function handleFetchError(error) {
288     const errorDisplayAlert = $('errorDisplay');
289     const errorMessage = "There was an error handling fetch request:";
290     if (!errorDisplayAlert) {
291         console.log(errorMessage, error);
292     } else {
293         errorDisplayAlert.innerHTML = errorMessage + "<br/>" + error;
294     }
295     return error;
296 }
