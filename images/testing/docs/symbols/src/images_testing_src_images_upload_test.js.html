 11 /**
 12  * Handles user uploads of files (typically images). 
 13  * Users can upload from their hard drive or provide a URL.
 14  * @constructor
 15  * @property {...File} sourceFileArray Files to be uploaded
 16  */
 17 function FileSource() {
 18     this.sourceFileArray = (() => {
 19         if ($('id_image_upload')) {
 20             return Array.from($('id_image_upload')).files;
 21         } else {
 22             return [];
 23         }

 49 /**
 50  * AJAX request for a presigned URL to be used for uploading a file to S3 bucket.
 51  * 
 52  * @param {File} sourceFile 
 53  * 
 54  * @returns {PresignedURLPacket} 
 55  * A {@link PresignedURLPacket} sent from AWS required for uploading file.
 56  */
 57 FileSource.prototype.getPresignedURLPacket = function(sourceFile) {
 58     return new Promise((res, rej) => {
 59         const url = document.querySelector('[name=sign_s3_url]').value
 60             + "?name=" + sourceFile.name + "type=" + sourceFile.type;
 61 
 62         makeFetch(url)
 63         .then((response) => {
 64             return response.json();
 65         })
 66         .then((json) => {
 67             // return the presigned url packet for uploading
 68             res(
 69                 {
 70                     file: sourceFile,       // File object
 71                     data: json.data,        // {} containing additional information 
 72                     url: json.image_url     // the URL of the file once it's uploaded
 73                 }
 74             );
 75         })
 76         .catch((error) => {
 77             rej(handleFetchError(error));
 78         });
 79     });
 80 }

 83 /**
 84  * Simple image retriever.
 85  * 
 86  * @source https://newbedev.com/how-to-convert-dataurl-to-file-object-in-javascript
 87  * 
 88  * @param {String} url 
 89  * Optional. Points to the image to be retrieved. Defaults to a random image API.
 90  * 
 91  * @param {String} fileName 
 92  * Optional. When uploaded to S3, filename gets randomized to a UUID hex.
 93  * 
 94  * @param {String} fileType
 95  * Optional. MIME-Type.
 96  * 
 97  * @returns {File}
 98  */
 99 FileSource.prototype.getImageFromURL = function(
100     url="https://picsum.photos/300",
101     fileName="image.jpg",
102     fileType="image/jpg") {
103     return new Promise((res, rej) => {
104         makeFetch(url)
105         .then((response) => {
106             let buffer = response.arrayBuffer();
107             let file = new File([buffer], fileName, {type: fileType});
108             res(file);
109         })
110         .catch((error) => {
111             rej(error);
112         })
113     })
114 }

132 /**
133  * Using this instead of the built-in fetch() API in order to provide better
134  * error-handling.
135  * 
136  * @param {String} url Points to the server path to use for a fetch() request
137  * @param {FetchOptions} options 
138  * @returns {Response}
139  */
140 function makeFetch(url, options={}) {
141     /*
142     options = { method:, body:, headers:, etc.} (from vanilla fetch() api)
143     */
144     return new Promise((res, rej) => {
145         fetch(url, options)
146         .then((response) => {
147             if (response.status >= 200 && response.status <= 299) {
148                 // request is ok, resolve response
149                 res(response);
150             }
151             else {
152                 throw new Error(response.statusText);
153             }
154         })
155         .catch((error) => {
156             rej(error);
157         })
158     });
159 }

162 /**
163  * Handles server error by either updating the UI or logging.
164  * @param {Error} error 
165  * @returns {Error}
166  */
167 function handleFetchError(error) {
168     const errorDisplayAlert = $('errorDisplay');
169     const errorMessage = "There was an error handling fetch request:";
170     if (!errorDisplayAlert) {
171         console.log(errorMessage, error);
172     } else {
173         errorDisplayAlert.innerHTML = errorMessage + "<br/>" + error;
174     }
175     return error;
176 }
